CREATE TABLE authors (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) UNIQUE NOT NULL,
    last_name VARCHAR(255) NOT NULL,
    first_name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    affiliation VARCHAR(255) NOT NULL
);

CREATE TABLE reviewers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) UNIQUE NOT NULL,
    last_name VARCHAR(255) NOT NULL,
    first_name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    affiliation VARCHAR(255) NOT NULL,
    verification_code(int) NOT NULL
);

CREATE TABLE editors (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) UNIQUE NOT NULL,
    last_name VARCHAR(255) NOT NULL,
    first_name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    affiliation VARCHAR(255) NOT NULL
);

CREATE TABLE readers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) UNIQUE NOT NULL,
    last_name VARCHAR(255) NOT NULL,
    first_name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,

CREATE TABLE article (
    article_id VARCHAR(20) PRIMARY KEY,   -- Unique identifier for each article
    title VARCHAR(255) NOT NULL,           -- Title of the article
    category VARCHAR(100),                 -- Category of the article
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP, -- Timestamp of when the article was created
    author VARCHAR(255)                   -- Action options: View/Edit/Delete
);

CREATE TABLE author_articles (
    id VARCHAR(50) NOT NULL PRIMARY KEY,

-- Create the sessions table
CREATE TABLE sessions (
    session_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    session_token VARCHAR(255) NOT NULL,
    login_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    logout_time TIMESTAMP NULL,
    FOREIGN KEY (user_id) REFERENCES authors(id) ON DELETE CASCADE
);
